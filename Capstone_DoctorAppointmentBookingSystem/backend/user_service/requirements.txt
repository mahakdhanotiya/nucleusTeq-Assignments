# =============================================================================
# user_service/requirements.txt
# =============================================================================
# This file lists every Python package the User Service needs to run.
# Install all of them at once with:  pip install -r requirements.txt
# =============================================================================

# --- Web framework ---
fastapi==0.115.0          # The web framework we use to build our API
uvicorn[standard]==0.30.6 # The server that actually runs our FastAPI app

# --- Database (MongoDB) ---
motor==3.7.1                # Async MongoDB driver (talks to MongoDB without blocking)
beanie==1.27.0              # ODM (Object Document Mapper) - lets us use MongoDB with Python classes

# --- Settings / environment variables ---
pydantic-settings==2.5.2    # Loads settings (like DB URL, secret keys) from a .env file
python-dotenv==1.0.1        # Reads the .env file itself

# --- Authentication & security ---
pyjwt==2.9.0                 # Creates and verifies JWT tokens (login tokens)
passlib[bcrypt]==1.7.4        # Hashes passwords securely using BCrypt
bcrypt==4.0.1                 # The actual BCrypt hashing algorithm (passlib uses this under the hood)

# --- Data validation ---
pydantic[email]==2.9.2        # Validates request/response data (includes EmailStr support)

# --- Testing (used later, included now so the environment is ready) ---
pytest==8.3.3                 # Test runner
pytest-mock==3.14.0            # Easy mocking for tests
pytest-asyncio==0.24.0          # Lets pytest run our async functions in tests
httpx==0.27.2                   # Used by FastAPI's TestClient to simulate API calls in tests
